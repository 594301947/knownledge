加法操作，一定要有3个元素，两个操作数a,b，一个进位carray

在a+b+carray之后，可能出现(a+b+carray) > 0的情况，要进行下面处理

1. 因为一个位置只能保存一个数，所以，要取个位数，即(a+b+carray) %10
2. 进位值carray要更新（carray = carray/10），用于下面场景
   - 用于下一次相加: a+b+carray
   - 所有位置都相加结束后，还要处理carray>0的情况

----

%10 计算低位（一个数字）

/ 10 计算除了低位剩下的高位

##### 题2: 两个链表相加

```c++
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* head = new ListNode();  // 创建一个“多余”的链表头节点
        ListNode* cur = head; // 当前指针，需要用它插入新节点，指向插入位置的前驱

        ListNode* p = l1;
        ListNode* q = l2;

        int carray = 0;
        while(p || q){
            int pval = p ? p->val : 0;  // 很关键: 如果p/q不存在，就将值设为0
            int qval = q ? q->val : 0;
            int sum = pval + qval + carray;

            carray = sum / 10;  // 更新进位值carray

            // 插入新值
            ListNode* cval = new ListNode(sum % 10);
            cur->next = cval;
            cur = cur->next;

            if (p)  // 只有p/q不为NULL，才向后移动
                p = p->next;
            if (q)
                q = q->next;
        }

        if (carray > 0){
            ListNode* cval = new ListNode(carray);
            cur->next = cval;
        }

        return head->next;
    }
};
```

class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        vector<int> ret;

        int m = digits.size();

        int carray = 0;
        carray++; // 因为是+1, 所以初始值carray+1
        for(int i=m-1; i>=0; i--){
            int sum = digits[i]+carray;
            carray = sum/10;
            digits[i] = sum%10;
        }
        if(carray > 0){
            ret.push_back(carray);
        }
        for(auto it : digits){
            ret.push_back(it);
        }

        return ret;
    }
};

##### 字符串相乘

```c++
class Solution {
public:
    string multiply(string num1, string num2) {
        int m = num1.size();
        int n = num2.size();

        // 结果最多为m+n位
        vector<int> vec(m+n);

        // 从个位数开始逐位相乘
        for (int i=m-1; i>=0; i--){
            for (int j=n-1; j>=0; j--){
                int mul = (num1[i]-'0') * (num2[j]-'0');
                
                // p,q: 乘积在vec对应的索引位置
                // p是低位, q是高位
                int p = i+j+1, q = i+j;

                int sum = mul + vec[p];
                vec[p] = sum % 10;   // 更新低位 = 最低位
                vec[q] += sum / 10;  // 更新高位 = vec[q] + 除最低位的其他位
            }
        }

        // "0000123" ---> "123"

        string ret = "";
        // 寻找第一个不为0的数的下标
        int i=0;
        for (; i<m+n; i++){
            if(vec[i] != 0)
                break;
        }
        // 将剩下的数全部保存到结果ret中
        for(;i<m+n;i++){
            ret += vec[i] + '0';
        }
        return ret == "" ? "0" : ret;
    }
};
```

##### 题9: 回文数

下面的解法，既高效，又可以避免整数溢出的风险

```c
bool isPalindrome(int x){
    int xT = 0;

    if (x > 0 && x % 10 == 0)  // 10 特殊数字判断: 大于0 && 末尾数字为0 (一定不是回文数)
        return false;

    while(x > xT){
        xT = xT * 10 + x % 10;
        x /= 10;
    }
    
    if (x == xT)  // 相等, 1221, 偶数个数
        return true;
    else{  // 不相等, 121, 一定是 x < xT 退出的
        if (x == xT / 10)
            return true;
        else
            return false;
    }
    return false;
}
```

##### 题8: atoi

```c++
class Solution {
public:
    bool isNum(char ch) {
        return ch >= '0' && ch <= '9';
    }
    int myAtoi(string str) {
        int len = str.size();
        
        int i = 0;
        
        // 去除左边的空格
        for (; i < len; i++){
            if(str[i]!=' '){
                break;
            }
        }

        // 1. 判断第一个字符
        // 1.1. 符号?
        int flag = 1;
        if(str[i] == '-'){
            flag = -1;
            i++;
        }
        else if(str[i] == '+'){
            i++;
        }
        if(!isNum(str[i])){ // 1.2. 第一个字符不是数字, 返回失败, 0
            return 0;
        }

        int ans = 0;
        for(; i<len;i++){
            if(!isNum(str[i]))
                break;

            // 更新ans之前，先判断越界
            if(ans > (INT_MAX - (str[i]-'0')) / 10)
                return flag == 1 ? INT_MAX : INT_MIN ;
            ans = ans*10 + (str[i]-'0');
        }
        return ans * flag;
    }
};
```
